Data Loading Example

import pandas as pd from sklearn.model_selection import train_test_split # Load datasets real_news = pd.read_csv('data/True.csv') fake_news = pd.read_csv('data/Fake.csv') # Add labels real_news['label'] = 1 # Real fake_news['label'] = 0 # Fake # Combine datasets dataset = pd.concat([real_news, fake_news], ignore_index=True) # Split for training X_train, X_test, y_train, y_test = train_test_split( dataset['text'], dataset['label'], test_size=0.2, random_state=42 )
